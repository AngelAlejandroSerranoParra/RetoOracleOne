const inputTextArea = document.getElementById("inputTextArea"); const copyButton = document.getElementById("copyButton"); copyButton.addEventListener('click', () => { // Seleccionar el contenido del textarea inputTextArea.select(); // Copiar el contenido seleccionado al portapapeles document.execCommand('copy'); // Des-seleccionar el textarea window.getSelection().removeAllRanges(); // Alerta al usuario que se copió el contenido alert('¡El contenido se copió al portapapeles!'); });
